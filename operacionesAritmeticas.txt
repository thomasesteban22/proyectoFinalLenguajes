# Operaciones aritméticas básicas

# Asignación de variables
a = 10;
b = 5;

# Suma y resta
suma = a + b;
resta = a - b;

# Multiplicación y división
multiplicacion = a * b;
division = a / b;

# Módulo y potencia
modulo = a % b;
potencia = a ^ b;

# Operaciones trigonométricas
angulo = 45;
radianes = angulo * (3.1416 / 180);
seno = sin(radianes);
coseno = cos(radianes);
tangente = tan(radianes);

# Raíz cuadrada
raiz = sqrt(a);

# Imprimir resultados
print(suma);
print(resta);
print(multiplicacion);
print(division);
print(modulo);
print(potencia);
print(seno);
print(coseno);
print(tangente);
print(raiz);
